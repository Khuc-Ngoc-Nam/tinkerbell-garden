```
cd server

```

## Bước 3: Cấu hình kết nối (Database Connection)

### 1. Tạo file `.env` trong thư mục `server/`:

```
PORT=5000
DB_HOST=host.docker.internal
DB_USER=root
DB_PASSWORD=mật_khẩu_mysql_của_bạn
DB_NAME=tinkerbell_db
DB_PORT=3306
```

### 2. Tạo file server/src/config/db.js:

```
const mysql = require('mysql2/promise');
require('dotenv').config();

const pool = mysql.createPool({
  host: process.env.DB_HOST,
  user: process.env.DB_USER,
  password: process.env.DB_PASSWORD,
  database: process.env.DB_NAME,
  port: process.env.DB_PORT,
  waitForConnections: true,
  connectionLimit: 10,
  queueLimit: 0
});

// Test thử kết nối ngay khi khởi động
pool.getConnection()
  .then(connection => {
    console.log("✅ Đã kết nối thành công tới MySQL Database!");
    connection.release();
  })
  .catch(err => {
    console.error("❌ Lỗi kết nối MySQL:", err.message);
  });

module.exports = pool;
```

## Sửa lại `server/src/modules/portal/portal.controller.js` như đã nói trong file "Lưu lại sau này sửa.md":

```
// server/src/modules/portal/portal.controller.js
const db = require('../../config/db');

// UC1.1: Lấy thông tin (Giả sử bạn query từ bảng Facility và TicketType)
const getParkInfo = async (req, res) => {
    try {
        // Tuỳ theo cấu trúc database.sql của bạn, tôi viết query mẫu:
        const [facilities] = await db.query('SELECT FacilityName FROM Facility WHERE Status = "Normal"');
        const [tickets] = await db.query('SELECT TypeName, BasePrice FROM TicketType');

        const parkInfo = {
            name: "TinkerBell Garden",
            description: "Khu vui chơi trẻ em hàng đầu với nhiều tiện ích.",
            ticketPrices: tickets.map(t => ({ type: t.TypeName, price: t.BasePrice })),
            facilities: facilities.map(f => f.FacilityName)
        };
        
        res.status(200).json({ success: true, data: parkInfo });
    } catch (error) {
        console.error("Lỗi getParkInfo:", error);
        res.status(500).json({ success: false, message: "Lỗi Server" });
    }
};

// UC1.2: Lấy danh sách sự kiện từ bảng EventCampaign
const getEvents = async (req, res) => {
    try {
        // Lấy các sự kiện chưa kết thúc
        const [events] = await db.query(
            'SELECT EventID as id, EventName as name, Description as description, StartDate as startDate, EndDate as endDate, DiscountRate as discountRate FROM EventCampaign WHERE EndDate >= CURDATE()'
        );
        
        res.status(200).json({ success: true, data: events });
    } catch (error) {
        console.error("Lỗi getEvents:", error);
        res.status(500).json({ success: false, message: "Lỗi Server" });
    }
};

module.exports = { getParkInfo, getEvents };
```


# Thêm UC1.3, UC1.4

## Bước 1: Thêm logic ghi dữ liệu vào Backend

Bác mở file `server/src/modules/portal/portal.controller.js`, cuộn xuống dưới cùng và thêm 2 hàm mới này vào (ngay trên dòng `module.exports = ...`).

```
// UC1.3: Đăng ký tham gia sự kiện
const bookEvent = async (req, res) => {
    try {
        const { eventId, fullName, email, phone } = req.body;
        
        // 1. Kiểm tra khách hàng đã tồn tại chưa (check theo SĐT)
        const [existingCustomer] = await db.query('SELECT CustomerID FROM Customer WHERE Phone = ?', [phone]);
        let customerId;
        
        if (existingCustomer.length > 0) {
            customerId = existingCustomer[0].CustomerID;
        } else {
            // Nếu chưa có thì tạo khách hàng mới
            customerId = 'CUS_' + Date.now();
            await db.query(
                'INSERT INTO Customer (CustomerID, FullName, Email, Phone, IsVIP) VALUES (?, ?, ?, ?, false)', 
                [customerId, fullName, email, phone]
            );
        }

        // 2. Tạo bản ghi Booking sự kiện
        const bookingId = 'BKG_' + Date.now();
        const qrCode = 'QR_' + bookingId; // Sinh mã QR ảo để test
        
        await db.query(
            'INSERT INTO EventBooking (BookingID, EventID, CustomerID, BookingDate, QRCode, Status) VALUES (?, ?, ?, NOW(), ?, "Confirmed")',
            [bookingId, eventId, customerId, qrCode]
        );

        res.status(200).json({ success: true, message: "Đăng ký sự kiện thành công!", qrCode });
    } catch (error) {
        console.error("Lỗi bookEvent:", error);
        res.status(500).json({ success: false, message: "Lỗi hệ thống khi đăng ký sự kiện" });
    }
};

// UC1.4: Đăng ký thẻ VIP Online
const registerVip = async (req, res) => {
    try {
        const { fullName, email, phone } = req.body;
        
        const [existingCustomer] = await db.query('SELECT CustomerID, IsVIP FROM Customer WHERE Phone = ?', [phone]);
        
        if (existingCustomer.length > 0) {
            if (existingCustomer[0].IsVIP) {
                return res.status(400).json({ success: false, message: "Khách hàng này đã là VIP từ trước!" });
            }
            // Update thành VIP, hạn 1 năm
            await db.query(
                'UPDATE Customer SET IsVIP = true, VIPExpiryDate = DATE_ADD(NOW(), INTERVAL 1 YEAR) WHERE Phone = ?', 
                [phone]
            );
        } else {
            // Tạo khách VIP mới
            const customerId = 'CUS_' + Date.now();
            await db.query(
                'INSERT INTO Customer (CustomerID, FullName, Email, Phone, IsVIP, VIPExpiryDate) VALUES (?, ?, ?, ?, true, DATE_ADD(NOW(), INTERVAL 1 YEAR))', 
                [customerId, fullName, email, phone]
            );
        }

        res.status(200).json({ success: true, message: "Đăng ký thẻ VIP thành công! Hạn dùng 1 năm." });
    } catch (error) {
        console.error("Lỗi registerVip:", error);
        res.status(500).json({ success: false, message: "Lỗi hệ thống khi đăng ký VIP" });
    }
};

// Cập nhật lại dòng export cuối file:
module.exports = { getParkInfo, getEvents, bookEvent, registerVip };
```

```
const express = require('express');

```
